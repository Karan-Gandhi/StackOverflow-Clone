- Back-end
    - Authentication
        - user
            - id -> string
            - displayName -> string
            - reputations -> Reputation[]
            - questions -> Question[]
            - answers -> Answers[]
            - profile pic -> string (url)
            - username -> string
            - password -> string

    - Reputation
        - userId -> user.id
        - type -> string (upvote, downvote)

    - Question
        - id -> string
        - title -> string
        - body -> string
        - edits -> Edit[]
        - userId -> user.id
        - upvotes -> Reputation[]
        - downvotes -> Reputation[]
        - status -> string (answered, open, closed, duplicate, needs to be focused)
        - reports -> Report[]
        - answers -> Answer[]
        - comments -> Comment[]

    - Edit
        - id -> string
        - userId -> user.id
        - body -> string
        - status -> string (accepted, rejected)

    - Report
        - id -> string
        - userId -> user.id
        - type -> string (dublicate, needs to be focused)

    - Answer
        - id -> string
        - userId -> user.id
        - body -> string
        - upvotes -> Reputation[]
        - downvotes -> Reputation[]
        - status -> string (normal, correct)
        - reports -> Report[]
        - edits -> Edit[]
        - comments -> Comment[]
    
    - Comment
        - body -> string
        - userId -> user.id
